function StatusBadge({ status }: { status: 'secure' | 'warning' | 'error' }) {
  const styles = {
    secure: 'bg-status-secure/10 text-status-secure border-status-secure/20',
    warning: 'bg-status-warning/10 text-status-warning border-status-warning/20',
    error: 'bg-status-error/10 text-status-error border-status-error/20',
  };
  const labels = {
    secure: 'Secured',
    warning: 'Incomplete',
    error: 'Action Required',
  };

  return (
    <span className={cn('rounded-full border px-2 py-0.5 text-[10px] font-mono font-bold uppercase tracking-wider', styles[status])}>
      {labels[status]}
    </span>
  );
}

function StatCard({ 
  icon: Icon, 
  label, 
  value, 
  description, 
  status, 
  progress 
}: { 
  icon: LucideIcon; 
  label: string; 
  value: number | string; 
  description: string;
  status: 'secure' | 'warning' | 'error';
  progress?: number;
}) {
  const displayValue = typeof value === 'number' ? value : 0;
  const springValue = useSpring(0, { stiffness: 60, damping: 20 });
  const rounded = useTransform(springValue, (latest) => Math.round(latest));
  
  useEffect(() => {
    springValue.set(displayValue);
  }, [displayValue, springValue]);

  return (
    <div className="surface-card p-6 text-left relative overflow-hidden group">
      <div className="flex items-center gap-3 mb-4">
        <div className={cn(
          "flex h-10 w-10 items-center justify-center rounded-xl transition-transform duration-300 group-hover:scale-110",
          status === 'secure' ? "bg-status-secure/10 text-status-secure" : "bg-status-warning/10 text-status-warning"
        )}>
          <Icon className="h-5 w-5" />
        </div>
        <span className="text-sm font-medium text-text-secondary">{label}</span>
      </div>
      
      <div className="flex items-baseline gap-1 mb-1">
        <motion.span className="text-3xl font-serif text-text-primary">
          {typeof value === 'number' ? rounded : value}
        </motion.span>
      </div>
      
      <p className="text-xs text-text-tertiary mb-4">{description}</p>
      
      {progress !== undefined && (
        <div className="h-1 w-full bg-bg-elevated rounded-full overflow-hidden">
          <motion.div 
            initial={{ width: 0 }}
            animate={{ width: `${progress}%` }}
            transition={{ duration: 1, ease: EASE_OUT_EXPO }}
            className={cn(
              "h-full rounded-full",
              status === 'secure' ? "bg-status-secure" : "bg-status-warning"
            )}
          />
        </div>
      )}
    </div>
  );
}

export function Settings() {
  const location = useLocation();
  useEffect(() => {
    if (location.hash) {
      const hashSection = location.hash.slice(1) as SectionId;
      if (sections.some(s => s.id === hashSection)) {
        setActiveSection(hashSection);
      }
    }
  }, [location.hash]);
  const { user } = useAuthStore();
  const [activeSection, setActiveSection] = useState<SectionId>('overview');
  const [providers, setProviders] = useState<AIProvider[]>([]);
  const [isLoadingProviders, setIsLoadingProviders] = useState(true);
  const [isSaving, setIsSaving] = useState(false);
  const [removingId, setRemovingId] = useState<string | null>(null);
  const [providerLoadError, setProviderLoadError] = useState<string | null>(null);

  const [form, setForm] = useState<ProviderFormState>(defaultFormState);

  const [isAddModalOpen, setIsAddModalOpen] = useState(false);
  const [visibleKeys, setVisibleKeys] = useState<Set<string>>(new Set());
  const [healthStatuses, setHealthStatuses] = useState<Record<string, 'working' | 'invalid' | 'untested'>>({});
  const [isTesting, setIsTesting] = useState<Record<string, boolean>>({});

  const [mcpServers, setMCPServers] = useState<MCPServer[]>([]);
  const [isLoadingMCPServers, setIsLoadingMCPServers] = useState(true);
  const [mcpLoadError, setMCPLoadError] = useState<string | null>(null);
  const [expandedMCPType, setExpandedMCPType] = useState<MCPServerType | null>(null);
  const [savingMCPType, setSavingMCPType] = useState<MCPServerType | null>(null);
  const [togglingMCPId, setTogglingMCPId] = useState<string | null>(null);
  const [removingMCPId, setRemovingMCPId] = useState<string | null>(null);

  const [usage, setUsage] = useState<{ projects_count: number; generations_count: number } | null>(null);

  const mcpSectionRef = useRef<HTMLDivElement>(null);

  const [mcpForms, setMCPForms] = useState<Record<MCPServerType, Record<MCPFormFieldName, string>>>(() => {
    const state = {} as any;
    mcpCardDefinitions.forEach((def) => {
      state[def.type] = def.fields.reduce((acc, field) => {
        acc[field.key] = '';
        return acc;
      }, {} as any);
    });
    return state;
  });

  const loadProviders = async () => {
    setIsLoadingProviders(true);
    setProviderLoadError(null);
    try {
      const data = await getAIProviders();
      setProviders(data);
    } catch (err) {
      setProviderLoadError(err instanceof Error ? err.message : 'Failed to load AI providers');
    } finally {
      setIsLoadingProviders(false);
    }
  };

  const loadMCPServers = async () => {
    setIsLoadingMCPServers(true);
    setMCPLoadError(null);
    try {
      const data = await getMCPServers();
      setMCPServers(data);
    } catch (err) {
      setMCPLoadError(err instanceof Error ? err.message : 'Failed to load research tools');
    } finally {
      setIsLoadingMCPServers(false);
    }
  };

  const loadUsage = async () => {
    try {
      const res = await fetch('/api/settings/usage', {
        headers: { Authorization: `Bearer ${await user?.getIdToken()}` },
      });
      if (res.ok) {
        setUsage(await res.json());
      }
    } catch (err) {
      console.error('Failed to load usage stats');
    }
  };

  useEffect(() => {
    void loadProviders();
    void loadMCPServers();
    void loadUsage();
  }, [user]);

  const selectedProvider = providerOptions.find((o) => o.value === form.provider)!;

  const handleSaveProvider = async (e: FormEvent) => {
    e.preventDefault();
    if (isSaving) return;

    if (!form.apiKey.trim()) {
      toast.error('API Key is required');
      return;
    }

    if (form.provider === 'custom' && !form.baseUrl.trim()) {
      toast.error('Base URL is required for custom providers');
      return;
    }

    setIsSaving(true);
    try {
      await saveAIProvider({
        name: providerLabels[form.provider],
        provider: form.provider,
        apiKey: form.apiKey.trim(),
        baseUrl: form.baseUrl.trim() || undefined,
        model: form.model.trim() || undefined,
        isDefault: providers.length === 0,
      });
      toast.success('AI Provider added');
      setForm(defaultFormState);
      await loadProviders();
    } catch (err) {
      toast.error(err instanceof Error ? err.message : 'Failed to save provider');
    } finally {
      setIsSaving(false);
    }
  };

  const handleRemoveProvider = async (id: string) => {
    try {
      setRemovingId(id);
      await deleteAIProvider(id);
      setProviders((prev) => prev.filter((p) => p.id !== id));
      toast.success('AI Provider removed');
    } catch (err) {
      toast.error(err instanceof Error ? err.message : 'Failed to remove provider');
    } finally {
      setRemovingId(null);
    }
  };

  const handleSaveMCPServer = async (e: FormEvent, type: MCPServerType) => {
    e.preventDefault();
    setSavingMCPType(type);
    try {
      const payload = {
        serverType: type,
        name: type === 'custom' ? mcpForms[type].serverName : mcpCardDefinitions.find(d => d.type === type)!.label,
        config: mcpForms[type],
      } as SaveMCPServerPayload;
      await saveMCPServer(payload);
      toast.success('Research tool connected');
      setExpandedMCPType(null);
      await loadMCPServers();
    } catch (err) {
      toast.error(err instanceof Error ? err.message : 'Failed to connect tool');
    } finally {
      setSavingMCPType(null);
    }
  };

  const handleToggleMCPServer = async (server: MCPServer) => {
    try {
      setTogglingMCPId(server.id);
      const result = await toggleMCPServer(server.id);
      setMCPServers(prev => prev.map(s => s.id === server.id ? { ...s, is_active: result.is_active } : s));
      toast.success(result.is_active ? 'Research tool paused' : 'Research tool active');
    } catch (err) {
      toast.error(err instanceof Error ? err.message : 'Failed to toggle tool');
    } finally {
      setTogglingMCPId(null);
    }
  };

  const handleRemoveMCPServer = async (server: MCPServer) => {
    try {
      setRemovingMCPId(server.id);
      await deleteMCPServer(server.id);
      setMCPServers(prev => prev.filter(s => s.id !== server.id));
      toast.success('Research tool removed');
    } catch (err) {
      toast.error(err instanceof Error ? err.message : 'Failed to remove tool');
    } finally {
      setRemovingMCPId(null);
    }
  };

  const toggleKeyVisibility = (id: string) => {
    setVisibleKeys((prev) => {
      const next = new Set(prev);
      if (next.has(id)) next.delete(id);
      else next.add(id);
      return next;
    });
  };

  const handleTestConnection = async (provider: AIProvider) => {
    setIsTesting((prev) => ({ ...prev, [provider.id]: true }));
    try {
      const ok = await testAIProvider(provider.id);
      setHealthStatuses((prev) => ({
        ...prev,
        [provider.id]: ok ? 'working' : 'invalid',
      }));
      if (ok) {
        toast.success(`Connection to ${providerLabels[provider.provider]} is working!`);
      } else {
        toast.error(`Could not connect to ${providerLabels[provider.provider]}. Please check your key.`);
      }
    } catch (err) {
      setHealthStatuses((prev) => ({ ...prev, [provider.id]: 'invalid' }));
      toast.error('Connection test failed.');
    } finally {
      setIsTesting((prev) => ({ ...prev, [provider.id]: false }));
    }
  };

  const handleMCPFieldChange = (type: MCPServerType, key: MCPFormFieldName, value: string) => {
    setMCPForms(prev => ({
      ...prev,
      [type]: { ...prev[type], [key]: value }
    }));
  };

  const mcpServersByType = useMemo(() => {
    const map: Record<string, MCPServer> = {};
    mcpServers.forEach((s) => {
      map[s.server_type] = s;
    });
    return map;
  }, [mcpServers]);

  const getToolStatusText = (server: MCPServer) => {
    return server.is_active ? 'Active' : 'Paused';
  };

  const getToolStatusClasses = (server: MCPServer) => {
    return server.is_active
      ? 'border-status-secure/20 bg-status-secure/10 text-status-secure'
      : 'border-status-warning/20 bg-status-warning/10 text-status-warning';
  };

  return (
    <div className="flex flex-col font-ui min-h-0 container mx-auto">
      {/* --- Sidebar --- */}
      

      {/* --- Main Content --- */}
      <main className="flex-1 flex flex-col pt-6 pb-20">
        <header className="px-6 py-6 border-b border-border-default flex items-center justify-between mb-8">
          <div className="flex items-center gap-4">
            <h2 className="text-[20px] font-display font-bold text-text-primary">
              {sections.find(s => s.id === activeSection)?.label}
            </h2>
            <StatusBadge status={providers.length > 0 ? 'secure' : 'warning'} />
          </div>
          <div className="flex items-center gap-2 text-xs text-text-tertiary font-mono uppercase tracking-widest">
            <ShieldCheck className="h-3.5 w-3.5 text-status-secure" />
            Encrypted connection
          </div>
        </header>

        <div className="px-6 pb-20 w-full">
          <motion.div
            key={activeSection}
            variants={containerVariants}
            initial="hidden"
            animate="visible"
            className="space-y-10"
          >
            {/* Overview Section */}
            {activeSection === 'overview' && (
              <div className="space-y-8">
                <header className="text-left">
                  <h3 className="text-3xl font-display font-bold text-text-primary tracking-tight">Overview</h3>
                  <p className="mt-2 text-lg font-light text-text-secondary">Everything at a glance.</p>
                </header>

                <div className="grid gap-6 sm:grid-cols-3">
                  <StatCard
                    icon={Cpu}
                    label="AI Coverage"
                    value={providers.length}
                    description={providers.length > 0 ? 'Active providers' : 'No providers added'}
                    status={providers.length > 0 ? 'secure' : 'warning'}
                    progress={providers.length > 0 ? 100 : 0}
                  />
                  <StatCard
                    icon={Search}
                    label="Research Depth"
                    value={mcpServers.filter((s) => s.is_active).length}
                    description="Live tools connected"
                    status={mcpServers.some((s) => s.is_active) ? 'secure' : 'warning'}
                    progress={(mcpServers.filter(s => s.is_active).length / 3) * 100}
                  />
                  <StatCard
                    icon={ShieldCheck}
                    label="Workspace Readiness"
                    value="Ready"
                    description="All systems configured"
                    status="secure"
                    progress={90}
                  />
                </div>

                <div className="surface-card p-8">
                  <div className="section-label mb-6">Recent Usage</div>
                  <div className="grid gap-8 sm:grid-cols-2">
                    <div className="flex flex-col gap-2">
                      <div className="text-[13px] font-medium text-text-secondary uppercase tracking-widest">Active Projects</div>
                      <div className="text-4xl font-serif text-text-primary">{usage?.projects_count || 0}</div>
                    </div>
                    <div className="flex flex-col gap-2">
                      <div className="text-[13px] font-medium text-text-secondary uppercase tracking-widest">Total Generations</div>
                      <div className="text-4xl font-serif text-text-primary">{usage?.generations_count || 0}</div>
                    </div>
                  </div>
                </div>
              </div>
            )}

            {/* Account Section */}
            {activeSection === 'account' && (
              <div className="space-y-8">
                <header className="text-left">
                  <h3 className="text-3xl font-display font-bold text-text-primary tracking-tight">Account</h3>
                  <p className="mt-2 text-lg font-light text-text-secondary">Manage your personal settings.</p>
                </header>

                <div className="surface-card p-8">
                  <div className="flex flex-col sm:flex-row items-start justify-between gap-8">
                    <div className="flex gap-6 items-center">
                      <div className="h-20 w-20 rounded-3xl bg-bg-elevated border border-border-default flex items-center justify-center overflow-hidden">
                        {user?.photoURL ? (
                          <img src={user.photoURL} alt="" className="h-full w-full object-cover" />
                        ) : (
                          <UserIcon className="h-8 w-8 text-text-tertiary" />
                        )}
                      </div>
                      <div className="text-left">
                        <h4 className="text-2xl font-display font-bold text-text-primary">{user?.displayName || 'Builder'}</h4>
                        <div className="flex items-center gap-2 mt-1 text-text-secondary">
                          <Mail className="h-4 w-4" />
                          <span className="text-sm">{user?.email}</span>
                        </div>
                      </div>
                    </div>
                    <button onClick={() => void logout()} className="btn-danger h-10 px-6">
                      <LogOut className="h-4 w-4 mr-2" />
                      Sign out
                    </button>
                  </div>
                </div>

                <div className="rounded-2xl border border-accent-border bg-accent-primary-muted/5 p-8 flex items-center justify-between text-left">
                  <div className="flex gap-4 items-center">
                    <div className="h-12 w-12 rounded-2xl bg-bg-surface border border-accent-border flex items-center justify-center text-accent-primary">
                      <Sparkles className="h-6 w-6" />
                    </div>
                    <div>
                      <h4 className="font-semibold text-text-primary">Upgrade to Scrimble Pro</h4>
                      <p className="text-sm text-text-secondary mt-0.5">Unlock priority generation and team collaboration.</p>
                    </div>
                  </div>
                  <button className="btn-primary h-10 px-6">Coming Soon</button>
                </div>
              </div>
            )}

            {/* Builder Profile */}
            {activeSection === 'profile' && (
              <BuilderProfileSection />
            )}

            {/* AI Keys Section */}
            {activeSection === 'ai' && (
              <div className="space-y-8">
                <div className="surface-card p-8 font-sans">
                  <div className="mb-8 flex items-start justify-between gap-4 text-left">
                    <div>
                      <h2 className="text-2xl font-display font-bold tracking-tight text-text-primary">AI Keys</h2>
                      <p className="mt-2 text-text-secondary">
                        Connect the models you trust to power your build plans.
                      </p>
                    </div>
                    <button
                      onClick={() => setIsAddModalOpen(true)}
                      className="btn-primary"
                    >
                      <Plus className="h-4 w-4" />
                      Add Key
                    </button>
                  </div>

                  {providerLoadError ? (
                    <div className="mb-6 flex flex-col gap-4 rounded-xl border border-status-error/20 bg-status-error/5 p-6 sm:flex-row sm:items-center sm:justify-between text-left">
                      <div>
                        <p className="text-sm font-medium text-text-primary">Failed to refresh keys</p>
                        <p className="text-xs text-text-secondary mt-1">{providerLoadError}</p>
                      </div>
                      <button onClick={() => void loadProviders()} className="btn-danger h-9 px-4 text-xs font-semibold">
                        <RefreshCw className="h-3.5 w-3.5 mr-2" />
                        Retry
                      </button>
                    </div>
                  ) : null}

                  <div className="rounded-2xl border border-border-default bg-bg-elevated/40 flex flex-col overflow-hidden">
                    {isLoadingProviders ? (
                      <div className="p-12">
                        <SectionSkeleton count={2} />
                      </div>
                    ) : providers.length > 0 ? (
                      providers.map((provider, idx) => {
                        const Icon = providerIcons[provider.provider] || KeyRound;
                        const isVisible = visibleKeys.has(provider.id);
                        const health = healthStatuses[provider.id] || 'untested';
                        const testing = isTesting[provider.id];

                        return (
                          <div key={provider.id}>
                            <div className="px-8 py-6 flex flex-col gap-6 lg:flex-row lg:items-center lg:justify-between text-left">
                              <div className="flex gap-5 items-center flex-1 min-w-0">
                                <div className="flex h-14 w-14 shrink-0 items-center justify-center rounded-2xl border border-border-default bg-bg-surface relative shadow-sm">
                                  <Icon className="h-6 w-6 text-accent-primary" />
                                  <div className={cn(
                                    "absolute -top-1 -right-1 h-3.5 w-3.5 rounded-full border-2 border-bg-surface shadow-sm",
                                    health === 'working' ? "bg-status-secure" :
                                    health === 'invalid' ? "bg-status-error" : "bg-text-tertiary"
                                  )} />
                                </div>
                                <div className="flex-1 min-w-0">
                                  <div className="mb-1 flex flex-wrap items-center gap-3">
                                    <h3 className="text-[16px] font-semibold tracking-tight text-text-primary">
                                      {providerLabels[provider.provider]}
                                    </h3>
                                    {provider.is_default ? (
                                      <span className="rounded-full border border-accent-border bg-accent-primary-muted px-2 py-0.5 text-[9px] font-mono font-bold uppercase tracking-wider text-accent-primary">
                                        Primary
                                      </span>
                                    ) : null}
                                  </div>
                                  <div className="flex items-center gap-3">
                                    <p className="text-sm text-text-tertiary truncate font-mono tracking-tight">
                                      {isVisible ? "sk-••••••••••••" : "••••••••••••••••"}
                                    </p>
                                    <button
                                      onClick={() => toggleKeyVisibility(provider.id)}
                                      className="text-text-tertiary hover:text-accent-primary transition-colors"
                                    >
                                      {isVisible ? <EyeOff className="h-4 w-4" /> : <Eye className="h-4 w-4" />}
                                    </button>
                                  </div>
                                </div>
                              </div>

                              <div className="flex items-center gap-3 shrink-0">
                                <button
                                  onClick={() => void handleTestConnection(provider)}
                                  disabled={testing}
                                  className="btn-secondary h-10 px-4 text-xs font-semibold"
                                >
                                  {testing ? <Loader2 className="h-3.5 w-3.5 animate-spin mr-2" /> : null}
                                  {health === 'working' ? 'Re-test' : 'Test Connection'}
                                </button>
                                <button
                                  type="button"
                                  onClick={() => void handleRemoveProvider(provider.id)}
                                  disabled={removingId === provider.id}
                                  className="btn-danger-ghost h-10 px-3"
                                >
                                  {removingId === provider.id ? (
                                    <Loader2 className="h-4 w-4 animate-spin" />
                                  ) : (
                                    <Trash2 className="h-4 w-4" />
                                  )}
                                </button>
                              </div>
                            </div>
                            {idx < providers.length - 1 && <hr className="border-border-default" />}
                          </div>
                        );
                      })
                    ) : (
                      <div className="flex flex-col items-center justify-center gap-6 p-10 text-center">
                        <p className="text-sm text-text-tertiary">No AI keys saved yet.</p>
                        <div className="flex flex-wrap justify-center gap-3">
                          {['openai', 'anthropic', 'gemini', 'openrouter'].map((provider) => {
                            const Icon = providerIcons[provider as AIProviderType] || KeyRound;
                            return (
                              <button
                                key={provider}
                                onClick={() => {
                                  setForm((current) => ({
                                    ...current,
                                    provider: provider as AIProviderType,
                                    apiKey: '',
                                    model: '',
                                    baseUrl: '',
                                  }));
                                  setIsAddModalOpen(true);
                                }}
                                className="inline-flex items-center gap-2 rounded-lg border border-border-default bg-bg-surface px-4 py-2.5 text-sm font-medium text-text-secondary transition-colors hover:border-accent-primary hover:text-accent-primary"
                              >
                                <Icon className="h-4 w-4" />
                                {providerLabels[provider as AIProviderType]}
                              </button>
                            );
                          })}
                        </div>
                      </div>
                    )}
                  </div>

                  <div className="mt-10 rounded-2xl border border-accent-border bg-accent-primary-muted/5 p-8 flex gap-6 text-left shadow-inner">
                    <div className="flex h-12 w-12 shrink-0 items-center justify-center rounded-2xl bg-bg-surface border border-accent-border shadow-sm">
                      <Lock className="h-6 w-6 text-status-secure" />
                    </div>
                    <div>
                      <h4 className="text-lg font-semibold text-text-primary">Your security is priority</h4>
                      <p className="mt-1 text-[15px] leading-relaxed text-text-secondary font-light">
                        All keys are encrypted with AES-256 before storage in Cloudflare D1. Scrimble never stores them in plain text and only uses them to execute work on your projects.
                      </p>
                    </div>
                  </div>
                </div>
              </div>
            )}

            {/* Research Tools Section */}
            {activeSection === 'research' && (
              <div className="space-y-8">
                <div id="research-tools" className="surface-card p-8 font-sans">
                  <div className="mb-8 text-left">
                    <h2 className="text-2xl font-display font-bold tracking-tight text-text-primary">Research Tools</h2>
                    <p className="mt-2 text-text-secondary">
                      Connect tools that let Scrimble do deeper research when building your plan.
                    </p>
                  </div>

                  {mcpLoadError ? (
                    <div className="mb-6 flex flex-col gap-4 rounded-xl border border-status-error/20 bg-status-error/5 p-6 sm:flex-row sm:items-center sm:justify-between text-left">
                      <div>
                        <p className="text-sm font-medium text-text-primary">Failed to load tools</p>
                        <p className="text-xs text-text-secondary mt-1">{mcpLoadError}</p>
                      </div>
                      <button onClick={() => void loadMCPServers()} className="btn-danger h-9 px-4 text-xs font-semibold">
                        <RefreshCw className="h-3.5 w-3.5 mr-2" />
                        Retry
                      </button>
                    </div>
                  ) : null}

                  <div className="rounded-2xl border border-border-default bg-bg-elevated/40 flex flex-col overflow-hidden">
                    {isLoadingMCPServers ? (
                      <div className="p-12">
                        <SectionSkeleton count={3} />
                      </div>
                    ) : (
                      mcpCardDefinitions.map((card, idx) => {
                        const connectedServer = mcpServersByType[card.type];
                        const isConnected = Boolean(connectedServer);
                        const isExpanded = expandedMCPType === card.type;
                        const isSavingCardLocal = savingMCPType === card.type;

                        return (
                          <div key={card.type}>
                            <div className="px-8 py-8 flex flex-col gap-6 lg:flex-row lg:items-center lg:justify-between text-left">
                              <div className="flex gap-6 items-center flex-1 min-w-0">
                                <div className="flex h-14 w-14 shrink-0 items-center justify-center rounded-2xl border border-border-default bg-bg-surface shadow-sm">
                                  <card.icon className="h-6 w-6 text-accent-primary" />
                                </div>

                                <div className="flex-1 min-w-0">
                                  <div className="mb-1 flex flex-wrap items-center gap-3">
                                    <h3 className="text-[16px] font-semibold tracking-tight text-text-primary">
                                      {card.label}
                                    </h3>
                                    {card.recommended ? (
                                      <span className="rounded-full border border-accent-border bg-accent-primary-muted px-2.5 py-0.5 text-[9px] font-mono font-bold uppercase tracking-wider text-accent-primary">
                                        Recommended
                                      </span>
                                    ) : null}
                                  </div>
                                  <p className="text-sm text-text-tertiary line-clamp-1 font-light">
                                    {card.description}
                                  </p>
                                </div>
                              </div>

                              <div className="flex shrink-0 items-center gap-4">
                                {isConnected && connectedServer ? (
                                  <>
                                    <span className={cn('inline-flex items-center gap-2 rounded-full border px-3 py-1 text-[11px] font-bold uppercase tracking-wider font-mono', getToolStatusClasses(connectedServer))}>
                                      <span className={cn('h-1.5 w-1.5 rounded-full', connectedServer.is_active ? 'bg-status-secure' : 'bg-status-warning')} />
                                      {getToolStatusText(connectedServer)}
                                    </span>
                                    <button
                                      type="button"
                                      onClick={() => void handleToggleMCPServer(connectedServer)}
                                      disabled={togglingMCPId === connectedServer.id}
                                      className="btn-secondary h-10 px-4 text-xs font-semibold"
                                    >
                                      {togglingMCPId === connectedServer.id && <Loader2 className="h-3.5 w-3.5 animate-spin mr-2" />}
                                      {connectedServer.is_active ? 'Pause' : 'Resume'}
                                    </button>
                                    <button
                                      type="button"
                                      onClick={() => void handleRemoveMCPServer(connectedServer)}
                                      disabled={removingMCPId === connectedServer.id}
                                      className="btn-danger-ghost h-10 px-3"
                                    >
                                      {removingMCPId === connectedServer.id ? (
                                        <Loader2 className="h-4 w-4 animate-spin" />
                                      ) : (
                                        <Trash2 className="h-4 w-4" />
                                      )}
                                    </button>
                                  </>
                                ) : (
                                  <button
                                    type="button"
                                    onClick={() => setExpandedMCPType(isExpanded ? null : card.type)}
                                    className={cn(
                                      'h-10 px-6 text-xs font-semibold',
                                      card.recommended 
                                        ? 'btn-primary' 
                                        : 'btn-secondary'
                                    )}
                                  >
                                    {isExpanded ? 'Cancel' : 'Connect'}
                                  </button>
                                )}
                              </div>
                            </div>

                            <AnimatePresence>
                              {isExpanded && !isConnected && (
                                <motion.div
                                  initial={{ height: 0, opacity: 0 }}
                                  animate={{ height: 'auto', opacity: 1 }}
                                  exit={{ height: 0, opacity: 0 }}
                                  transition={{ duration: 0.3, ease: EASE_OUT_EXPO }}
                                  className="overflow-hidden"
                                >
                                  <div className="px-8 pb-8">
                                    <div className="rounded-2xl border border-border-default bg-bg-surface/50 p-8">
                                      <form onSubmit={(e) => void handleSaveMCPServer(e, card.type)} className="space-y-8">
                                        <div className="flex flex-col sm:flex-row sm:items-center sm:justify-between text-left">
                                          <div>
                                            <h4 className="text-lg font-semibold text-text-primary">Configure {card.label}</h4>
                                            {card.note && <p className="mt-1 text-sm text-text-tertiary">{card.note}</p>}
                                          </div>
                                          {card.getLinkHref && (
                                            <a href={card.getLinkHref} target="_blank" rel="noreferrer" className="inline-flex items-center gap-1.5 text-xs font-medium text-accent-soft hover:text-accent-primary">
                                              {card.getLinkLabel}
                                              <ExternalLink className="h-3 w-3" />
                                            </a>
                                          )}
                                        </div>

                                        <div className="grid gap-6 sm:grid-cols-2 text-left">
                                          {card.fields.map((field) => (
                                            <div key={field.key} className={card.fields.length === 1 ? 'sm:col-span-2' : ''}>
                                              <label className="mb-2 block text-xs font-bold uppercase tracking-wider text-text-tertiary">{field.label}</label>
                                              <input
                                                type={field.type}
                                                autoComplete={field.autoComplete}
                                                value={mcpForms[card.type][field.key]}
                                                onChange={(e) => handleMCPFieldChange(card.type, field.key, e.target.value)}
                                                placeholder={field.placeholder}
                                                className={inputClassName}
                                              />
                                              {field.helper && <p className="mt-2 text-[11px] text-text-muted leading-relaxed italic">{field.helper}</p>}
                                            </div>
                                          ))}
                                        </div>

                                        <div className="flex justify-end gap-3">
                                          <button type="submit" disabled={isSavingCardLocal} className="btn-primary px-8">
                                            {isSavingCardLocal && <Loader2 className="h-4 w-4 animate-spin mr-2" />}
                                            {isSavingCardLocal ? 'Connecting...' : 'Connect Tool'}
                                          </button>
                                        </div>
                                      </form>
                                    </div>
                                  </div>
                                </motion.div>
                              )}
                            </AnimatePresence>
                            {idx < mcpCardDefinitions.length - 1 && <hr className="border-border-default mx-8" />}
                          </div>
                        );
                      })
                    )}
                  </div>
                </div>
              </div>
            )}
          </motion.div>
        </div>
      </main>

      <Dialog open={isAddModalOpen} onOpenChange={setIsAddModalOpen}>
        <DialogContent className="max-w-xl">
          <DialogHeader>
            <DialogTitle>Add AI Provider</DialogTitle>
            <DialogDescription>
              Connect a new AI service to power your build plans.
            </DialogDescription>
          </DialogHeader>

          <form onSubmit={(e) => {
            handleSaveProvider(e).then(() => setIsAddModalOpen(false));
          }} className="space-y-8 mt-4">
            <div className="text-left">
              <label className="mb-2 block text-[13px] font-medium text-text-primary/50">
                Select Provider
              </label>
              <div className="grid gap-2 rounded-xl border border-border-default bg-bg-elevated p-1.5 sm:grid-cols-5">
                {providerOptions.map((option) => (
                  <button
                    key={option.value}
                    type="button"
                    onClick={() =>
                      setForm((current) => ({
                        ...current,
                        provider: option.value,
                        apiKey: '',
                        model: '',
                        baseUrl: '',
                      }))
                    }
                    className={cn(
                      'h-10 rounded-lg px-2 text-xs font-medium transition-all duration-200',
                      form.provider === option.value
                        ? 'bg-bg-surface text-accent-primary shadow-sm ring-1 ring-accent-border'
                        : 'text-text-secondary hover:bg-bg-surface/50 hover:text-text-primary'
                    )}
                  >
                    {option.label}
                  </button>
                ))}
              </div>
              <p className="mt-4 text-[13px] leading-relaxed text-text-primary/50 italic font-light">
                {selectedProvider.helper}
              </p>
            </div>

            <div className="grid gap-6 sm:grid-cols-2 text-left">
              <div className={cn(form.provider === 'custom' ? 'sm:col-span-2' : '')}>
                <label className="mb-2 block text-[13px] font-medium text-text-primary/50">
                  Your API key
                </label>
                <input
                  type="password"
                  autoComplete="current-password"
                  value={form.apiKey}
                  onChange={(event) =>
                    setForm((current) => ({ ...current, apiKey: event.target.value }))
                  }
                  placeholder={selectedProvider.placeholder}
                  className={inputClassName}
                />
              </div>

              {form.provider !== 'custom' && (
                <div>
                  <label className="mb-2 block text-[13px] font-medium text-text-primary/50">
                    Model to use (optional)
                  </label>
                  <input
                    type="text"
                    value={form.model}
                    onChange={(event) =>
                      setForm((current) => ({ ...current, model: event.target.value }))
                    }
                    placeholder="Provider default"
                    className={inputClassName}
                  />
                </div>
              )}

              {form.provider === 'custom' && (
                <div className="sm:col-span-2">
                  <label className="mb-2 block text-[13px] font-medium text-text-primary/50">
                    API Base URL
                  </label>
                  <input
                    type="url"
                    value={form.baseUrl}
                    onChange={(event) =>
                      setForm((current) => ({ ...current, baseUrl: event.target.value }))
                    }
                    placeholder="https://..."
                    className={inputClassName}
                  />
                </div>
              )}
            </div>

            <div className="flex justify-end gap-3">
              <button
                type="button"
                onClick={() => setIsAddModalOpen(false)}
                className="btn-secondary px-6"
              >
                Cancel
              </button>
              <button
                type="submit"
                disabled={isSaving}
                className="btn-primary px-8"
              >
                {isSaving ? <Loader2 className="h-4 w-4 animate-spin" /> : null}
                {isSaving ? 'Saving...' : 'Save AI Key'}
              </button>
            </div>
          </form>
        </DialogContent>
      </Dialog>
    </div>
  );
}

export default Settings;
